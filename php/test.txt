function handleSwipe() {
    let wrapper = document.getElementById('imagemap');
    let touchLength = 0;
    var myVar = setInterval(()=>{
    	wrapper.addEventListener('touchstart', (e) => {
    		console.log('Start===',e.touches.length)
    		touchLength = e.touches.length;
    		if(touchLength == 1) {
    			touchstartX = 0
    			touchendX = 0
    			touchstartX = e.changedTouches[0].screenX
    		}
    	});
    }, 1000);
	
	wrapper.addEventListener('touchend', (e) => {
		console.log('End===',e.touches.length)
		if(touchLength == 1 && e.touches.length == 0) {
		  	touchendX = e.changedTouches[0].screenX
		  	checkDirection()
		  	clearInterval(myVar);
		}
	});
}

checkDirection = () => {
	
    let horizontalDifference = touchendX - touchstartX;
    var midpoint = Math.floor(screen.width/2);
    if (touchendX < touchstartX && Math.abs(horizontalDifference) > SWPIE_THRESHOLD) {
    	if(index != (full_array.length - 1)) {
    		goToPage(index+1);
    	}
    }
    if (touchendX > touchstartX && Math.abs(horizontalDifference) > SWPIE_THRESHOLD) {
    	if(index != 0) {
    		goToPage(index-1);
    	}
    }
  }
